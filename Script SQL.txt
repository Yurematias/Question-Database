CREATE TABLE Domain
(
	dominio varchar(100),
	primary key(dominio)
);

CREATE TABLE Theme
(
	tema varchar(100),
	primary key(tema)
);

CREATE TABLE Question
(
	id_questao SERIAL NOT NULL,
	texto varchar(300),
	resposta varchar(300),
	dominio varchar(100),
	tema varchar(100),
	dificuldade integer,
	
	primary key(id_questao),
	foreign key(dominio) REFERENCES Domain(Dominio),
	foreign key(tema) REFERENCES Theme(tema)
)
